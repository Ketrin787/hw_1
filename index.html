git config --global user.name - задать имя
git config --global user.email - задать почту
git init - создать лок репориторий
git status - проверить текущий статус
git add - выбрать файлы
git commit - закомитить, как скопировать
git log - используется для просмотра истории коммитов
git checkout - вернуться назад или вернуться к версии
git branch - можно создавать, удалять, переименовывать ветки
git push - отправить изменения на удаленный репозиторий
git pull - стянуть изменения с удаленного репозитория
git merge - используется для слияния одной или нескольких веток в текущую
git clone - скопировать с удаленного репозитория
.gitignore файл - используется для автоматического игнорирования файлов опредеоенного типа
git revert - создает новую фиксацию с откатными изменениями
git reset - стирает вашу историю git вместо создания нового коммита
git rebase - повторить все изменения и с одной ветки на другую
git cherry-pick - заново применить изменения в виде нового коммита наверху текущей ветки
git stash - создание новой задачи
